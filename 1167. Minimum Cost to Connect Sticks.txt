*1167. Minimum Cost to Connect Sticks

Solution :

class Solution {
    public int connectSticks(int[] sticks) {
        PriorityQueue<Integer> pq = new PriorityQueue(); // by default min heap
        
        for(int stick : sticks)
            pq.offer(stick);
        
        int cost = 0;
        while(pq.size() > 1) {
            int first = pq.poll();
            int second = pq.poll();
            int combined = (first + second);
            
            cost += combined;
            pq.offer(combined);
        }
        
        return cost;
    }
}
