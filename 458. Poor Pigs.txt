*458. Poor Pigs

Solution : 

class Solution {
    public int poorPigs(int buckets, int minutesToDie, int minutesToTest) {
        if(buckets <= 1) return 0;
        int base = minutesToTest / minutesToDie + 1;
        
        int pigs = 0;
        while( (int) Math.pow(base, pigs) < buckets) {
            pigs++;
        }
        
        return pigs;
    }
}
