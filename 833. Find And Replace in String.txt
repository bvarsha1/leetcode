*833. Find And Replace in String

Solution : 

class Solution {
    public String findReplaceString(String s, int[] indices, String[] sources, String[] targets) {
        StringBuilder sb = new StringBuilder(s);
        
        // make an ordered map to get ordered list of indices to check and modify
        // there will be no overlap either
        TreeMap<Integer, Integer> indexMap = new TreeMap();
        for(int i = 0; i < indices.length; i++) {
            indexMap.put(indices[i], i);
        }
        
        // iterate over the map in descending order to avoid changing the index while replace
        for(int start : indexMap.descendingKeySet()) {
            int i = indexMap.get(start);
            int end = start + sources[i].length();
            if(s.substring(start, end).equals(sources[i]))
                sb.replace(start, end, targets[i]);
        }
        
        return sb.toString();
    }
}
